    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const passwordInput = document.getElementById('password-input');
            const submitButton = document.getElementById('submit-button');
            const passwordForm = document.getElementById('password-form');
            const galleryContainer = document.getElementById('gallery-container');
            const messageBox = document.getElementById('message-box');

            // The secret password
            const correctPassword = 'Rafayel';

            /**
             * Displays a message in the message box.
             * @param {string} message - The message to display.
             * @param {string} type - The type of message ('success' or 'error').
             */
            function showMessage(message, type) {
                messageBox.textContent = message;
                messageBox.className = ''; // Clear existing classes
                messageBox.classList.add('p-3', 'rounded-md', 'text-center', 'font-medium', 'mt-4', 'block');
                
                if (type === 'success') {
                    messageBox.classList.add('bg-green-100', 'text-green-700');
                } else if (type === 'error') {
                    messageBox.classList.add('bg-red-100', 'text-red-700');
                }

                // Hide the message after 3 seconds
                setTimeout(() => {
                    messageBox.classList.add('hidden');
                }, 3000);
            }

            /**
             * Handles the password submission.
             */
            function handlePasswordSubmit() {
                const enteredPassword = passwordInput.value;
                if (enteredPassword === correctPassword) {
                    showMessage('Success! You can now view the photos.', 'success');
                    // Hide the form and show the gallery after a delay
                    setTimeout(() => {
                        passwordForm.classList.add('hidden');
                        galleryContainer.classList.remove('hidden');
                    }, 500);
                } else {
                    showMessage('Incorrect password. Please try again.', 'error');
                    passwordInput.value = '';
                }
            }

            // Event listener for the button click
            submitButton.addEventListener('click', handlePasswordSubmit);

            // Event listener for the 'Enter' key press on the input field
            passwordInput.addEventListener('keydown', (e) => {
                if (e.key === 'Enter') {
                    handlePasswordSubmit();
                }
            });
        });
    </script>
